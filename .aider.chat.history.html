import markdown
import re

def parse_sections(markdown_file):
    # Read the markdown file
    with open(markdown_file, 'r', encoding='utf-8') as f:
        content = f.read()
    
    # Split content into sections based on top-level headings
    sections = re.split(r'^# ', content, flags=re.MULTILINE)[1:]
    
    # Create a dictionary with heading as key and content as value
    parsed_sections = {}
    for section in sections:
        # Get the heading (first line)
        heading = section.split('\n')[0].strip()
        # Get the content (everything after heading)
        content = '# ' + section  # Add back the # that was removed in split
        parsed_sections[heading] = content
    
    return parsed_sections

def generate_html_pages(sections):
    # Initialize markdown converter
    md = markdown.Markdown(extensions=['fenced_code', 'codehilite'])
    
    # Create HTML template
    html_template = """
    <!DOCTYPE html>
    <html>
    <head>
        <meta charset="utf-8">
        <title>{title}</title>
        <style>
            body {{ 
                max-width: 800px; 
                margin: 0 auto; 
                padding: 20px;
                font-family: Arial, sans-serif;
            }}
            pre {{ 
                background-color: #f5f5f5; 
                padding: 10px; 
                border-radius: 5px;
            }}
            code {{
                font-family: monospace;
            }}
        </style>
    </head>
    <body>
        {content}
        <hr>
        <footer>
            <p><a href="index.html">Back to Index</a></p>
        </footer>
    </body>
    </html>
    """
    
    # Generate index page content
    index_content = "<h1>Contents</h1>\n<ul>"
    for heading in sections.keys():
        file_name = heading.lower().replace(' ', '_') + '.html'
        index_content += f'<li><a href="{file_name}">{heading}</a></li>\n'
    index_content += "</ul>"
    
    # Create index page
    with open('index.html', 'w', encoding='utf-8') as f:
        f.write(html_template.format(title='Index', content=index_content))
    
    # Create individual pages for each section
    for heading, content in sections.items():
        file_name = heading.lower().replace(' ', '_') + '.html'
        html_content = md.convert(content)
        
        with open(file_name, 'w', encoding='utf-8') as f:
            f.write(html_template.format(title=heading, content=html_content))

def main():
    # Parse the markdown file
    sections = parse_sections('notes.md')
    
    # Generate HTML pages
    generate_html_pages(sections)
    
    print("HTML pages generated successfully!")

if __name__ == "__main__":
    main()

import re
import os
import subprocess
from datetime import datetime

def split_markdown_file(input_file, output_dir):
    # Create output directory if it doesn't exist
    if not os.path.exists(output_dir):
        os.makedirs(output_dir)
    
    # Read the markdown file
    with open(input_file, 'r', encoding='utf-8') as f:
        content = f.read()
    
    # Split content into sections based on top-level headings
    sections = re.split(r'^# ', content, flags=re.MULTILINE)[1:]
    
    # Write each section to a separate file
    for section in sections:
        # Get the heading (first line)
        heading = section.split('\n')[0].strip()
        # Create filename from heading
        filename = heading.lower().replace(' ', '_').replace('(', '').replace(')', '') + '.md'
        filepath = os.path.join(output_dir, filename)
        
        # Write content to file with heading
        with open(filepath, 'w', encoding='utf-8') as f:
            f.write('# ' + section)
        
        # Convert to HTML using pandoc
        output_html = filepath.replace('.md', '.html')
        
        pandoc_cmd = [
            'pandoc',
            filepath,
            '-o', output_html,
            '--standalone',
            '--template=./default.html5',
            '--css', 'https://latex.now.sh/style.css',
            '--include-before-body=topbar.html',
            '--include-after-body=dark.js',
            '--include-after-body=license.html',
            '--metadata', 'title=ecallen',
            '--variable=lastUpdated:' + str(int(datetime.now().timestamp())),
            '--variable=creationDate:' + str(int(datetime.now().timestamp()))
        ]
        
        subprocess.run(pandoc_cmd)
        print(f"Generated {output_html}")

def main():
    input_file = 'notes.md'
    output_dir = 'sections'
    split_markdown_file(input_file, output_dir)
    print("All sections processed successfully!")

if __name__ == "__main__":
    main()
